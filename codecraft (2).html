Code

One SDK, Every Language

Deploy, monitor, and manage your apps with our native SDKs. Consistent APIs across all languages.

// Deploy a full-stack app in seconds
import { CodeCraft } from '@codecraft/sdk';

const app = new CodeCraft({
  name: 'my-awesome-app',
  template: 'nextjs',
  region: 'global',
  env: { 'NODE_ENV': 'production' }
});

// Auto-scaling, SSL, CDN included
await app.deploy();

// Stream live logs
app.on('log', (log) => console.log(log));
// Full type safety end-to-end
import { CodeCraft, Config } from '@codecraft/sdk';

interface AppConfig extends Config {
  aiEnabled: boolean;
  cache: 'lru' | 'ttl';
}

const config: AppConfig = {
  name: 'typed-app',
  template: 'nextjs',
  aiEnabled: true,
  cache: 'ttl'
};

const app = new CodeCraft<AppConfig>(config);
await app.deploy();
# Python SDK for CodeCraft
from codecraft import CodeCraft, Config

class MyConfig(Config):
  name: str = "python-api"
  template: str = "fastapi"
  workers: int = 4

config = MyConfig()
app = CodeCraft(config)

# Deploy with auto-scaling
app.deploy()

# Monitor in real-time
for log in app.stream_logs():
  print(log)
// Zero-allocation Go SDK
package main

import (
  "context"
  cc "github.com/codecraft/go-sdk"
)

func main() {
  ctx := context.Background()

  app := cc.New(cc.Config{
    Name: "go-service",
    Template: "gin",
    Workers: 8,
  })

  // Deploy with one call
  if err := app.Deploy(ctx); err != nil {
    panic(err)
  }
}
// Memory-safe Rust SDK
use codecraft::{CodeCraft, Config};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
  let config = Config {
    name: "rust-api".to_string(),
    template: "axum".to_string(),
    workers: 16,
  };

  let app = CodeCraft::new(config);
  app.deploy().await?;

  println!("Deployed!");
  Ok(())
}
# Elegant Ruby SDK
require 'codecraft'

config = CodeCraft::Config.new(
  name: "ruby-app",
  template: "rails",
  workers: 4
)

app = CodeCraft.new(config)

# Deploy with one method call
app.deploy

# Stream logs with blocks
app.stream_logs do |log|
  puts log
end

FAQ

Frequently Asked Questions

Everything you need to know. Can't find your answer? Our support team is here to help.

What frameworks does CodeCraft support?

CodeCraft supports virtually every modern framework including React, Vue, Angular, Svelte, Next.js, Nuxt, Astro, Remix, and more. Backend frameworks include Express, Fastify, Django, FastAPI, Ruby on Rails, Laravel, and Go-based frameworks.

Is there a free tier available?

Yes! Our Starter plan is completely free forever. You get 5 projects, 200GB bandwidth, community support, basic analytics, and automatic SSL. Perfect for side projects, portfolios, and learning.

How does the AI copilot work?

Our AI copilot analyzes your entire codebase in real-time to provide context-aware suggestions. It understands your project structure, dependencies, and coding patterns to generate relevant completions, refactor suggestions, and bug predictions.

Can I use my own custom domain?

Custom domains are available on Pro and Enterprise plans. Add your domain in the dashboard, update your DNS records, and we handle SSL provisioning automatically with zero configuration.

What happens if I exceed my bandwidth limit?

We notify you when you approach your limit. You can upgrade your plan or purchase additional bandwidth at $0.08 per GB. Your projects will never go offline unexpectedly.

Is my data secure on CodeCraft?

Absolutely. We're SOC 2 Type II certified, ISO 27001 compliant, and GDPR ready. All data is encrypted with AES-256 at rest and TLS 1.3 in transit. We never access your source code.
